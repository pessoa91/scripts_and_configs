% My personal latex commands

\newcommand{\HGC}[2]{
	\begin{center}
	\includegraphics[height=#1\textheight]{#2}
	\end{center}
	}

\newcommand{\WGC}[2]{
	\begin{center}
	\includegraphics[width=#1\textwidth]{#2}
	\end{center}
	}

\newcommand{\introframe}[1]{
    \begin{frame}
        \centering
        \Huge #1
    \end{frame}
}

\newcommand{\introsection}[1]{
    \section{#1}
    \introframe{#1}
}

\newcommand{\derivative}[2]{
    \frac{\text{d}#1}{\text{d}#2}
}
